```bash
function clean_deps
{
    folders=("node_modules" "venv" ".env" "vendor")
    days=${1:-90}

    for folder in "${folders[@]}"
    do
         find ~/dev -name $folder -type d -maxdepth 4 -mtime +$days -exec rm -rf {} \;
    done
}
```
